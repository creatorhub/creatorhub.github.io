// ========================================================= // PREMIUM LEAFLET MAP & REVERSE GEOCLICK ENGINE // ========================================================= let liveMapInstance = null; let liveMarkerInstance = null; function initializeSmartMap() { const mapBox = document.getElementById("mapContainerBlock"); // अगर दोबारा क्लिक करें तो मैप बंद/चालू हो (Toggle) if (mapBox.style.display === "block") { mapBox.style.display = "none"; return; } mapBox.style.display = "block"; if (!liveMapInstance) { // डिफ़ॉल्ट फोकस अहमदाबाद/गुजरात पर सेट करें [23.0225, 72.5714] liveMapInstance = L.map('mapContainerBlock').setView([23.0225, 72.5714], 12); // साफ और प्रीमियम ओपन-स्ट्रीट टाइल्स लोड करें L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { maxZoom: 19, attribution: '© CreatorHub Maps' }).addTo(liveMapInstance); // 🗺️ GPS ऑटो-डिटेक्ट: यूजर से लाइव लोकेशन की परमिशन मांगें if (navigator.geolocation) { navigator.geolocation.getCurrentPosition(function(pos) { let lat = pos.coords.latitude; let lng = pos.coords.longitude; liveMapInstance.setView([lat, lng], 16); // सटीक घर पर ज़ूम करें dropMapPin(lat, lng); reverseGeocodeAddress(lat, lng); }, function() { // अगर यूजर GPS ब्लॉक कर दे, तो डिफ़ॉल्ट पिन शो करें dropMapPin(23.0225, 72.5714); }); } // मैप पर कहीं भी क्लिक करने पर पिन सरकाएं liveMapInstance.on('click', function(e) { dropMapPin(e.latlng.lat, e.latlng.lng); reverseGeocodeAddress(e.latlng.lat, e.latlng.lng); }); } else { // मैप रेंडरिंग ग्लिच फिक्स (Displacement Fix) setTimeout(() => { liveMapInstance.invalidateSize(); }, 250); } } function dropMapPin(lat, lng) { if (liveMarkerInstance) { liveMarkerInstance.setLatLng([lat, lng]); } else { liveMarkerInstance = L.marker([lat, lng], { draggable: true }).addTo(liveMapInstance); // पिन को घसीटने (Drag) के बाद भी एड्रेस बदल जाए liveMarkerInstance.on('dragend', function() { let position = liveMarkerInstance.getLatLng(); reverseGeocodeAddress(position.lat, position.lng); }); } } // 🌟 Nominatim API के ज़रिए अक्षांश से असली इंडियन एड्रेस निकालना async function reverseGeocodeAddress(lat, lng) { try { let response = await fetch(`https://nominatim.openstreetmap.org/reverse?format=jsonv2&lat=${lat}&lon=${lng}&accept-language=en`); let data = await response.json(); if (!data || !data.address) return; let addr = data.address; // 1. शहर (City) ऑटो-फिल let detectedCity = addr.city || addr.town || addr.village || addr.suburb || ""; if (document.getElementById("city")) { document.getElementById("city").value = detectedCity; } // 2. राज्य (State) ऑटो-फिल if (document.getElementById("state")) { document.getElementById("state").value = addr.state || ""; } // 3. पिनकोड (Pincode) ऑटो-फिल if (document.getElementById("pincode")) { document.getElementById("pincode").value = addr.postcode || ""; } // 4. मुख्य एड्रेस बिल्डिंग/रोड लाइन ऑटो-फिल if (document.getElementById("address")) { let apartment = addr.house_number || addr.building || addr.residential || ""; let roadName = addr.road || addr.neighbourhood || ""; let localArea = addr.suburb || addr.county || ""; let fullLine = `${apartment} ${roadName} ${localArea}`.replace(/\s+/g, ' ').trim(); document.getElementById("address").value = fullLine; } } catch (err) { console.log("Map Reverse Code Error: ", err); } }
